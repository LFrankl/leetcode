func countLargestGroup(n int) int {
    count := make(map[int]int)
    maxSize := 0

    for i := 1; i <= n; i++ {
        sum := 0
        num := i
        for num > 0 {
            sum += num % 10
            num /= 10
        }
        count[sum]++
        if count[sum] > maxSize {
            maxSize = count[sum]
        }
    }

    result := 0
    for _, v := range count {
        if v == maxSize {
            result++
        }
    }
    return result
}

class Solution {
public:
    int countLargestGroup(int n) {
        unordered_map<int, int> count;
        int maxSize = 0;

        for (int i = 1; i <= n; i++) {
            int sum = 0, num = i;
            while (num > 0) {
                sum += num % 10;
                num /= 10;
            }
            count[sum]++;
            if (count[sum] > maxSize) {
                maxSize = count[sum];
            }
        }

        int result = 0;
        for (auto& [_, v] : count) {
            if (v == maxSize) {
                result++;
            }
        }
        return result;
    }
};
