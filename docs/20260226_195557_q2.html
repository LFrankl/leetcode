    sum1, sum2 := 0, 0
    for _, num := range nums1 {
        sum1 += num
    }
    for _, num := range nums2 {
        sum2 += num
    }

    // ç¡®ä¿ sum1 >= sum2ï¼Œæ–¹ä¾¿å¤„ç†
    if sum1 < sum2 {

    }

    diff := sum1 - sum2
    if diff == 0 {
        return 0
    }

    // æ”¶é›†å˜åŒ–é‡
    changes := make([]int, 0)

    // nums1 ä¸­çš„å…ƒç´ å¯ä»¥å‡å°‘ï¼ˆå› ä¸º sum1 æ›´å¤§ï¼‰

        int sum2 = accumulate(nums2.begin(), nums2.end(), 0);

        // ç¡®ä¿ sum1 >= sum2

class Solution {
public:
    int minOperations(vector<int>& nums1, vector<int>& nums2) {
        int sum1 = accumulate(nums1.begin(), nums1.end(), 0);
        int sum2 = accumulate(nums2.begin(), nums2.end(), 0);

        // ç¡®ä¿ sum1 >= sum2
        if (sum1 < sum2) {
            swap(nums1, nums2);
            swap(sum1, sum2);
        }

        int diff = sum1 - sum2;
        if (diff == 0) return 0;

        // æ’åº
        sort(nums1.begin(), nums1.end());  // å‡åº
        sort(nums2.begin(), nums2.end());  // å‡åº

        int operations = 0;
        int i = nums1.size() - 1, j = 0;

        while (diff > 0 && (i >= 0 || j < nums2.size())) {
            int reduce = (i >= 0) ? nums1[i] - 1 : 0;
            int increase = (j < nums2.size()) ? 6 - nums2[j] : 0;

            if (reduce >= increase) {
                diff -= reduce;
                i--;
            } else {
                diff -= increase;
                j++;
            }
            operations++;

            if (diff <= 0) {
                return operations;
            }
        }

        return -1;
    }
};
