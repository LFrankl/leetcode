Hopper 公司查询 I（LeetCode 1635）

简要思路

这是一道数据库 SQL 题，核心是对 2020 年的 12 个月生成统计报告：每月的活跃司机数（该月及之前加入的司机累计数）和接受订单数（该月内完成的订单数）。关键难点在于需要保证输出 12 行（即使某月数据为 0），以及正确计算累计活跃司机数。

解法一：Go 程序化处理（模拟 SQL 逻辑）

思路：用 Go 结构体模拟三张表，遍历 1-12 月，分别统计累计活跃司机和当月接受订单数。活跃司机是累计值（join_date ≤ 当月末的司机总数），接受订单是当月值。

package main

import (
    "fmt"
    "time"
)

// Drivers 表
type Driver struct {
    DriverID int
    JoinDate time.Time
}

// AcceptedRides 表
type AcceptedRide struct {
    RideID             int
    DriverID           int
    RideDate           time.Time
    RideDurationMinutes int
}

// 输出行
type ResultRow struct {
    Month         int
    ActiveDrivers int
    AcceptedRides int
}

func hopperCompanyQueriesI(drivers []Driver, rides []AcceptedRide) []ResultRow {
    results := make([]ResultRow, 12)

    // 预处理：按月统计接受订单数
    monthlyRides := make(map[int]int) // month -> count
    for _, r := range rides {
        month := int(r.RideDate.Month())
        year := r.RideDate.Year()
        if year == 2020 && month >= 1 && month <= 12 {
            monthlyRides[month]++
        }
    }

    for m := 1; m <= 12; m++ {
        // 计算当月末日期
        endOfMonth := time.Date(2020, time.Month(m)+1, 1, 0, 0, 0, 0, time.UTC).Add(-time.Nanosecond)

        // 统计活跃司机：join_date <= 当月末
        activeCount := 0
        for _, d := range drivers {
            if !d.JoinDate.After(endOfMonth) {
                activeCount++
            }
        }

        results[m-1] = ResultRow{
            Month:         m,
            ActiveDrivers: activeCount,
            AcceptedRides: monthlyRides[m],
        }
    }

    return results
}

func main() {

        {DriverID: 5,  JoinDate: time.Date(2020, 2, 16, 0, 0, 0, 0, time.UTC)},

        {DriverID: 4,  JoinDate: time.Date(2020, 5, 17, 0, 0, 0, 0, time.UTC)},
        {DriverID: 1,  JoinDate: time.Date(2020, 7, 8, 0, 0, 0, 0, time.UTC)},
        {DriverID: 6,  JoinDate: time.Date(2020, 11, 2, 0, 0, 0, 0, time.UTC)},
    }

    rides := []AcceptedRide{
        {RideID: 6, DriverID: 10, RideDate: time.Date(2020, 3, 16, 0, 0, 0, 0, time.UTC), RideDurationMinutes: 63},
        {RideID: 1, DriverID: 8,  RideDate: time.Date(2020, 3, 19, 0, 0, 0, 0, time.UTC), RideDurationMinutes: 14},
        {RideID: 7, DriverID: 4,  RideDate: time.Date(2020, 6, 5, 0, 0, 0, 0, time.UTC), RideDurationMinutes: 29},
        {RideID: 2, DriverID: 1,  RideDate: time.Date(2020, 7, 22, 0, 0, 0, 0, time.UTC), RideDurationMinutes: 67},

复杂度：时间 O(12 × D + R)（D 为司机数，R 为订单数），空间 O(12 + R)

解法二：预排序 + 二分查找优化活跃司机统计

思路：先将司机按加入日期排序，对每月末日期用二分查找快速定位活跃司机数量（joindate ≤ monthend 的累计数），避免每月重新遍历所有司机。

package main

import (
    "fmt"
    "sort"
    "time"
)

type Driver struct {
    DriverID int
    JoinDate time.Time
}

type AcceptedRide struct {
    RideID              int
    DriverID            int
    RideDate            time.Time
    RideDurationMinutes int
}

type ResultRow struct {
    Month         int
    ActiveDrivers int
    AcceptedRides int
}

func hopperCompanyQueriesI(drivers []Driver, rides []AcceptedRide) []ResultRow {
    // 按 JoinDate 排序
    sort.Slice(drivers, func(i, j int) bool {
        return drivers[i].JoinDate.Before(drivers[j].JoinDate)
    })

    // 预处理：按月统计订单数
    monthlyRides := make(map[int]int)
    for _, r := range rides {
        m := int(r.RideDate.Month())
        if r.RideDate.Year() == 2020 && m >= 1 && m <= 12 {
            monthlyRides[m]++
        }
    }

    results := make([]ResultRow, 12)
    for m := 1; m <= 12; m++ {
        // 当月末日期
        endOfMonth := time.Date(2020, time.Month(m)+1, 1, 0, 0, 0, 0, time.UTC).Add(-time.Nanosecond)

        // 二分查找：找最后一个 JoinDate <= endOfMonth 的位置
        idx := sort.Search(len(drivers), func(i int) bool {
            return drivers[i].JoinDate.After(endOfMonth)
        })
        // idx 即为活跃司机数量（前 idx 个司机都满足条件）

        results[m-1] = ResultRow{
            Month:         m,
            ActiveDrivers: idx,
            AcceptedRides: monthlyRides[m],
        }
    }

    return results
}

func main() {
    drivers := []Driver{
        {10, time.Date(2019, 12, 10, 0, 0, 0, 0, time.UTC)},
        {8, time.Date(2020, 1, 13, 0, 0, 0, 0, time.UTC)},
        {5, time.Date(2020, 2, 16, 0, 0, 0, 0, time.UTC)},
        {7, time.Date(2020, 3, 8, 0, 0, 0, 0, time.UTC)},
        {4, time.Date(2020, 5, 17, 0, 0, 0, 0, time.UTC)},
        {1, time.Date(2020, 7, 8, 0, 0, 0, 0, time.UTC)},
        {6, time.Date(2020, 11, 2, 0, 0, 0, 0, time.UTC)},
    }

    rides := []AcceptedRide{
        {6, 10, time.Date(2020, 3, 16, 0, 0, 0, 0, time.UTC), 63},
        {1, 8, time.Date(2020, 3, 19, 0, 0, 0, 0, time.UTC), 14},
        {7, 4, time.Date(2020, 6, 5, 0, 0, 0, 0, time.UTC), 29},
        {2, 1, time.Date(2020, 7, 22, 0, 0, 0, 0, time.UTC), 67},
        {3, 7, time.Date(2020, 10, 16, 0, 0, 0, 0, time.UTC), 8},
    }

    results := hopperCompanyQueriesI(drivers, rides)
    for _, r := range results {
        fmt.Printf("Month: %d, ActiveDrivers: %d, AcceptedRides: %d\n", r.Month, r.ActiveDrivers, r.AcceptedRides)
    }
}

复杂度：时间 O(D log D + 12 log D + R)（排序 + 12次二分查找 + 订单统计），空间 O(12 + R)

对比：解法二在司机数量大时显著优于解法一，因为每月活跃司机统计从 O(D) 降到 O(log D)；若数据量小则解法一更简洁直观。
