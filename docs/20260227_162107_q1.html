func findEvenNumbers(digits []int) []int {
    n := len(digits)
    seen := make(map[int]bool)

    for i := 0; i < n; i++ {
        if digits[i] == 0 {
            continue // ç™¾ä½ä¸èƒ½ä¸º0
        }
        for j := 0; j < n; j++ {
            if i == j {
                continue
            }
            for k := 0; k < n; k++ {
                if k == i || k == j {
                    continue
                }
                if digits[k] % 2 != 0 {
                    continue // ä¸ªä½å¿…é¡»å¶æ•°
                }
                num := digits[i]*100 + digits[j]*10 + digits[k]
                seen[num] = true
            }
        }
    }

    res := make([]int, 0, len(seen))
    for num := range seen {
        res = append(res, num)
    }
    sort.Ints(res)
    return res
}

func findEvenNumbers(digits []int) []int {
    freq := make([]int, 10)
    for _, d := range digits {
        freq[d]++
    }

    resMap := make(map[int]bool)

    var dfs func(pos int, num int, used []int)
    dfs = func(pos int, num int, used []int) {
        if pos == 3 {
            if num % 2 == 0 && num >= 100 {
                resMap[num] = true
            }
            return
        }

        start := 0
        if pos == 0 {
            start = 1 // ç™¾ä½ä¸èƒ½ä¸º0
        }

        for d := start; d <= 9; d++ {
            if used[d] < freq[d] {
                used[d]++
                dfs(pos+1, num*10 + d, used)
                used[d]--
            }
        }
    }

    used := make([]int, 10)
    dfs(0, 0, used)

    res := make([]int, 0, len(resMap))
    for num := range resMap {
        res = append(res, num)
    }
    sort.Ints(res)
    return res
}

class Solution {
public:
    void dfs(int pos, int num, vector<int>& freq, vector<int>& used, unordered_set<int>& res) {
        if (pos == 3) {
            if (num % 2 == 0 && num >= 100) {
                res.insert(num);
            }
            return;
        }

        int start = (pos == 0) ? 1 : 0;
        for (int d = start; d <= 9; d++) {
            if (used[d] < freq[d]) {
                used[d]++;
                dfs(pos + 1, num * 10 + d, freq, used, res);
                used[d]--;
            }
        }
    }

    vector<int> findEvenNumbers(vector<int>& digits) {
        vector<int> freq(10, 0);
        for (int d : digits) {
            freq[d]++;
        }

        unordered_set<int> resSet;
        vector<int> used(10, 0);
        dfs(0, 0, freq, used, resSet);

        vector<int> res(resSet.begin(), resSet.end());
        sort(res.begin(), res.end());
        return res;
    }
};
