## AI 解答 (DeepSeek)

### 解法一：哈希集合过滤 + 按日期分组统计

思路：先用 Users 表构建被禁止用户的哈希集合，然后筛选出行程中乘客和司机都未被禁止且日期在指定范围内的记录，按日期分组统计总行程数和取消行程数，最后计算取消率并四舍五入保留两位小数。

```go
package main

import (
	"fmt"
	"math"
	"sort"
)

type Trip struct {
	ID        int
	ClientID  int
	DriverID  int
	CityID    int
	Status    string
	RequestAt string
}

type User struct {
	UsersID int
	Banned  string
	Role    string
}

type Result struct {
	Day             string
	CancellationRate float64
}

func cancellationRate(trips []Trip, users []User) []Result {
	// 构建被禁止用户的哈希集合
	bannedSet := make(map[int]bool)
	for _, u := range users {
		if u.Banned == "Yes" {
			bannedSet[u.UsersID] = true
		}
	}

	// 按日期分组统计
	dayStats := make(map[string]struct{ total, cancelled int })

	for _, t := range trips {
		// 日期范围过滤
		if t.RequestAt < "2013-10-01" || t.RequestAt > "2013-10-03" {
			continue
		}
		// 乘客和司机都必须未被禁止
		if bannedSet[t.ClientID] || bannedSet[t.DriverID] {
			continue
		}

		stats := dayStats[t.RequestAt]
		stats.total++
		if t.Status == "cancelled_by_driver" || t.Status == "cancelled_by_client" {
			stats.cancelled++
		}
		dayStats[t.RequestAt] = stats
	}

	// 构建结果
	results := make([]Result, 0, len(dayStats))
	for day, stats := range dayStats {
		rate := float64(0)
		if stats.total > 0 {
			rate = math.Round(float64(stats.cancelled)/float64(stats.total)*100) / 100
		}
		results = append(results, Result{Day: day, CancellationRate: rate})
	}

	// 按日期排序输出（题目说无顺序要求，排序仅为美观）
	sort.Slice(results, func(i, j int) bool {
		return results[i].Day < results[j].Day
	})
	return results
}

func main() {
	trips := []Trip{
		{1, 1, 10, 1, "completed", "2013-10-01"},
		{2, 2, 11, 1, "cancelled_by_driver", "2013-10-01"},
		{3, 3, 12, 6, "completed", "2013-10-01"},
		{4, 4, 13, 6, "cancelled_by_client", "2013-10-01"},
		{5, 1, 10, 1, "completed", "2013-10-02"},
		{6, 2, 11, 6, "completed", "2013-10-02"},
		{7, 3, 12, 6, "completed", "2013-10-02"},
		{8, 2, 12, 12, "completed", "2013-10-03"},
		{9, 3, 10, 12, "completed", "2013-10-03"},
		{10, 4, 13, 12, "cancelled_by_driver", "2013-10-03"},
	}

	users := []User{
		{1, "No", "client"},
		{2, "Yes", "client"},
		{3, "No", "client"},
		{4, "No", "client"},
		{10, "No", "driver"},
		{11, "No", "driver"},
		{12, "No", "driver"},
		{13, "No", "driver"},
	}

	results := cancellationRate(trips, users)
	for _, r := range results {
		fmt.Printf("| %s | %.2f |\n", r.Day, r.CancellationRate)
	}
}
```

复杂度：时间 O(T + U)，空间 O(U + D)，其中 T 为行程数，U 为用户数，D 为日期数。构建哈希集合遍历 Users O(U)，过滤统计遍历 Trips O(T)，每个行程只做哈希查找 O(1)。

### 解法二：一次遍历 + 内联判断

思路：与解法一类似，但不做预构建哈希集合，而是将 Users 表转化为 `map[int]string`（users_id → banned），在遍历行程时直接查 map 判断乘客和司机是否被禁止。本质上复杂度一致，只是组织方式不同，更贴近 SQL 的 JOIN 语义。

```go
package main

import (
	"fmt"
	"math"
	"sort"
)

type Trip struct {
	ID        int
	ClientID  int
	DriverID  int
	CityID    int
	Status    string
	RequestAt string
}

type User struct {
	UsersID int
	Banned  string
	Role    string
}

type Result struct {
	Day              string
	CancellationRate float64
}

func cancellationRate(trips []Trip, users []User) []Result {
	// 构建 users_id -> banned 的映射，模拟 SQL JOIN
	bannedMap := make(map[int]string)
	for _, u := range users {
		bannedMap[u.UsersID] = u.Banned
	}

	// 按日期聚合统计
	type stats struct { total, cancelled int }
	dayStats := make(map[string]stats)

	for _, t := range trips {
		if t.RequestAt < "2013-10-01" || t.RequestAt > "2013-10-03" {
			continue
		}
		// 乘客和司机都必须 banned == "No"，模拟两次 JOIN Users 表
		if bannedMap[t.ClientID] == "Yes" || bannedMap[t.DriverID] == "Yes" {
			continue
		}

		s := dayStats[t.RequestAt]
		s.total++
		if t.Status != "completed" {
			s.cancelled++
		}
		dayStats[t.RequestAt] = s
	}

	results := []Result{}
	for day, s := range dayStats {
		rate := 0.0
		if s.total > 0 {
			rate = math.Round(float64(s.cancelled)/float64(s.total)*100) / 100
		}
		results = append(results, Result{day, rate})
	}

	sort.Slice(results, func(i, j int) bool { return results[i].Day < results[j].Day })
	return results
}

func main() {
	trips := []Trip{
		{1, 1, 10, 1, "completed", "2013-10-01"},
		{2, 2, 11, 1, "cancelled_by_driver", "2013-10-01"},
		{3, 3, 12, 6, "completed", "2013-10-01"},
		{4, 4, 13, 6, "cancelled_by_client", "2013-10-01"},
		{5, 1, 10, 1, "completed", "2013-10-02"},
		{6, 2, 11, 6, "completed", "2013-10-02"},
		{7, 3, 12, 6, "completed", "2013-10-02"},
		{8, 2, 12, 12, "completed", "2013-10-03"},
		{9, 3, 10, 12, "completed", "2013-10-03"},
		{10, 4, 13, 12, "cancelled_by_driver", "2013-10-03"},
	}
	users := []User{
		{1, "No", "client"}, {2, "Yes", "client"}, {3, "No", "client"},
		{4, "No", "client"}, {10, "No", "driver"}, {11, "No", "driver"},
		{12, "No", "driver"}, {13, "No", "driver"},
	}
	for _, r := range cancellationRate(trips, users) {
		fmt.Printf("| %s | %.2f |\n", r.Day, r.CancellationRate)
	}
}
```

复杂度：时间 O(T + U)，空间 O(U + D)。与解法一相同，核心瓶颈在于遍历行程表和构建用户映射，哈希查找每次 O(1)。

---

**补充说明**：此题本质是 SQL 问题（LeetCode 262），等价 SQL 为：

```sql
SELECT t.request_at AS Day,
       ROUND(SUM(IF(t.status != 'completed', 1, 0)) / COUNT(*), 2) AS CancellationRate
FROM Trips t
JOIN Users u1 ON t.client_id = u1.users_id AND u1.banned = 'No'
JOIN Users u2 ON t.driver_id = u2.users_id AND u2.banned = 'No'
WHERE t.request_at BETWEEN '2013-10-01' AND '2013-10-03'
GROUP BY t.request_at;
```

Go 解法模拟了上述 SQL 的逻辑：两次 JOIN → 两次哈希查找判断 banned，GROUP BY → map 按日期分组，ROUND → `math.Round`。
